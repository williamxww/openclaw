## 1. 整体架构

```
CronServiceContract（接口）
  src/cron/service-contract.ts
        ↓
CronService（门面）
  src/cron/service.ts
        ↓
ops.ts（CRUD + start/stop）  ←→  timer.ts（定时器 + 执行）
        ↓                              ↓
  store.ts（持久化）          jobs.ts（调度计算）
        ↓
  isolated-agent/run.ts（隔离 Agent 执行）
        ↓
```

**两种任务类型：**

| 类型 | `sessionTarget` | `payload.kind` | 执行方式 |
|------|----------------|----------------|---------|
| 主 Session 任务 | `"main"` | `"systemEvent"` | 向主 Session 注入系统事件，通过 heartbeat 唤醒 |
| 隔离 Agent 任务 | `"isolated"` / `"current"` | `"agentTurn"` | 启动独立 Agent 实例执行，结果可投递到渠道 |

**文件：** `src/cron/service/ops.ts:163`

```typescript
export async function start(state: CronServiceState) {  // ops.ts:163
  // 1. 加载持久化 store
  await locked(state, async () => {
    await ensureLoaded(state, { skipRecompute: true });
    // 2. 扫描被中断的任务（runningAtMs 非空 = 上次 gateway 重启时正在运行）
    for (const job of state.store?.jobs ?? []) {
      if (typeof job.state.runningAtMs === "number") {
        markInterruptedStartupRun({ state, job, ... });
        // 标记为 error，consecutiveErrors++，清除 runningAtMs
      }
    }
    await persist(state);
  });

  // 3. 补跑启动前错过的任务（含 stagger 逻辑）
  await runMissedJobs(state, {
    skipJobIds: interruptedJobIds,
    deferAgentTurnJobs: true,  // agentTurn 任务延迟 2 分钟，避免占用 channel connect 窗口
  });

  // 4. 修复 nextRunAtMs，启动定时器
  await locked(state, async () => {
    recomputeNextRunsForMaintenance(state);
    await persist(state);
    armTimer(state);  // 启动定时器
  });
}
```

## 4. 定时器管理（armTimer / onTimer）

### armTimer

**文件：** `src/cron/service/timer.ts:1118`

```typescript
export function armTimer(state: CronServiceState) {  // timer.ts:1118
  if (state.timer) clearTimeout(state.timer);
  state.timer = null;

  if (!state.deps.cronEnabled) return;

  const nextAt = nextWakeAtMs(state);  // 所有 job 中最近的 nextRunAtMs
  if (!nextAt) {
    // 有 enabled job 但无 nextRunAtMs → 60s 维护检查
    armRunningRecheckTimer(state);
    return;
  }

  const now = state.deps.nowMs();
  const delay = Math.max(nextAt - now, 0);

  // 关键：delay=0 时强制最小 2s，防止 stuck runningAtMs 导致热循环（#17821）
  const flooredDelay = delay === 0 ? MIN_REFIRE_GAP_MS : delay;  // MIN_REFIRE_GAP_MS = 2_000
  // 最大 60s，防止时钟漂移
  const clampedDelay = Math.min(flooredDelay, MAX_TIMER_DELAY_MS);  // MAX_TIMER_DELAY_MS = 60_000

  state.timer = setTimeout(() => {
    void onTimer(state).catch(...);
  }, clampedDelay);
}
```

### onTimer

**文件：** `src/cron/service/timer.ts:1187`

```typescript
export async function onTimer(state: CronServiceState) {  // timer.ts:1187
  if (state.running) {
    // 上一个 job 还在跑 → re-arm 60s 检查，不阻塞调度器（#12025）
    armRunningRecheckTimer(state);
    return;
  }
  state.running = true;
  armRunningRecheckTimer(state);  // watchdog：执行期间保持定时器活跃

  try {
    const dueJobs = await locked(state, async () => {
      await ensureLoaded(state, { forceReload: true, skipRecompute: true });
      const due = collectRunnableJobs(state, state.deps.nowMs());  // timer.ts:1444

      if (due.length === 0) {
        // 无到期任务 → 仅维护 nextRunAtMs，不执行
        recomputeNextRunsForMaintenance(state, { recomputeExpired: true });
        await persist(state);
        return [];
      }

      // 标记所有到期任务为 running
      for (const job of due) {
        job.state.runningAtMs = state.deps.nowMs();
        job.state.lastError = undefined;
      }
      await persist(state);
      return due;
    });

    // 并发执行（受 maxConcurrentRuns 限制）
    await runDueJobs(state, dueJobs);
  } finally {
    state.running = false;
    armTimer(state);  // 执行完毕，重新计算下次唤醒时间
  }
}
```

## 5. 调度时间计算

**文件：** `src/cron/schedule.ts:68`

```typescript
export function computeNextRunAtMs(schedule: CronSchedule, nowMs: number): number | undefined {
  // ── at：一次性定时 ──────────────────────────────────────────────
  if (schedule.kind === "at") {
    const atMs = parseAbsoluteTimeMs(schedule.at);
    return atMs > nowMs ? atMs : undefined;  // 已过期则返回 undefined
  }

  // ── every：固定间隔 ─────────────────────────────────────────────
  if (schedule.kind === "every") {
    const everyMs = Math.max(1, Math.floor(schedule.everyMs));
    const anchor = schedule.anchorMs ?? nowMs;
    const elapsed = nowMs - anchor;
    const steps = Math.max(1, Math.ceil(elapsed / everyMs));
    return anchor + steps * everyMs;
  }

  // ── cron：标准 Cron 表达式（使用 croner 库）─────────────────────
  const cron = resolveCachedCron(schedule.expr, resolveCronTimezone(schedule.tz));
  let next = cron.nextRun(new Date(nowMs));
  // 处理 croner 的时区年份回退 bug：若返回过去时间，从明天重试
  if (next && next.getTime() <= nowMs) {
    next = cron.nextRun(new Date(nowMs + 1000)) ?? cron.nextRun(new Date(tomorrowMs));
  }
  return next?.getTime();
}
```

```typescript
function computeStaggeredCronNextRunAtMs(job: CronJob, nowMs: number) {  // jobs.ts:86
  if (job.schedule.kind !== "cron") return computeNextRunAtMs(job.schedule, nowMs);

  const staggerMs = resolveCronStaggerMs(job.schedule);
  const offsetMs = resolveStableCronOffsetMs(job.id, staggerMs);  // jobs.ts:65
  if (offsetMs <= 0) return computeNextRunAtMs(job.schedule, nowMs);

  // 将 cursor 向前移 offsetMs，找到 base next，再加回 offsetMs
  // 确保 shifted 时间 > nowMs
  let cursorMs = Math.max(0, nowMs - offsetMs);
  for (let attempt = 0; attempt < 4; attempt++) {
    const baseNext = computeNextRunAtMs(job.schedule, cursorMs);
    if (!baseNext) return undefined;
    const shifted = baseNext + offsetMs;
    if (shifted > nowMs) return shifted;
    cursorMs = Math.max(cursorMs + 1, baseNext + 1_000);
  }
  return undefined;
}
```

**确定性 stagger 偏移（SHA-256）：**

**文件：** `src/cron/service/jobs.ts:65`

```typescript
function resolveStableCronOffsetMs(jobId: string, staggerMs: number) {  // jobs.ts:65
  if (staggerMs <= 1) return 0;
  const cacheKey = `${staggerMs}:${jobId}`;
  const cached = staggerOffsetCache.get(cacheKey);
  if (cached !== undefined) return cached;

  // 对 jobId 做 SHA-256，取前 4 字节 % staggerMs
  // 同一 jobId 每次重启偏移不变（确定性）
  const digest = crypto.createHash("sha256").update(jobId).digest();
  const offset = digest.readUInt32BE(0) % staggerMs;
  staggerOffsetCache.set(cacheKey, offset);
  return offset;
}
```

```typescript
export async function executeJobCoreWithTimeout(  // timer.ts:159
  state: CronServiceState,
  job: CronJob,

  const controller = new AbortController();

    clearTimeout(timer);
```

```typescript
export async function executeJob(  // timer.ts:1923
  state: CronServiceState,
  job: CronJob,
  _nowMs: number,
  _opts: { forced: boolean },
) {
  const startedAt = state.deps.nowMs();
  job.state.runningAtMs = startedAt;
  markCronJobActive(job.id);
  emit(state, { jobId: job.id, action: "started", job, runAtMs: startedAt });

  let coreResult;
  try {
    coreResult = await executeJobCoreWithTimeout(state, job);
  } catch (err) {
    coreResult = { status: "error", error: String(err) };
  }

  const endedAt = state.deps.nowMs();
  // 写回结果：更新 lastRunStatus、nextRunAtMs、consecutiveErrors 等
  const shouldDelete = applyJobResult(state, job, { status, error, startedAt, endedAt });

  emitJobFinished(state, job, coreResult, startedAt);

  // at 类型任务执行后自动删除
  if (shouldDelete && state.store) {
    state.store.jobs = state.store.jobs.filter((j) => j.id !== job.id);
    emit(state, { jobId: job.id, action: "removed", job });
  }
  clearCronJobActive(job.id);
}
```

### Watchdog 机制

三阶段超时保护：

```
setup watchdog (60s)
  ↓ runner 启动后清除
pre-execution watchdog (min(60s, timeout/2))
  ↓ 进入 execution 阶段后清除
主 timeout（job.payload.timeoutSeconds 或默认值）
```

通过 `onExecutionPhase` 回调跟踪执行阶段，阶段进入 `execution` 后清除 pre-execution watchdog。

## 10. 错开策略（Stagger）

OpenClaw 有**两种独立的 stagger 机制**，解决不同场景的并发问题。

### 10.1 启动时错过任务的 stagger

**文件：** `src/cron/service/timer.ts:1480`

当 gateway 重启后发现多个错过的任务：

```typescript
async function planStartupCatchup(state, opts) {  // timer.ts:1480
  const maxImmediate = state.deps.maxMissedJobsPerRestart ?? 5;  // 默认最多立即执行 5 个

  const missed = collectRunnableJobs(state, now, { skipAtIfAlreadyRan: true });
  const sorted = missed.toSorted((a, b) => a.state.nextRunAtMs - b.state.nextRunAtMs);

  // agentTurn 任务单独延迟（避免占用 channel connect 窗口）
  const deferredAgentJobs = sorted.filter(job => job.payload.kind === "agentTurn");
  const startupEligible = sorted.filter(job => job.payload.kind !== "agentTurn");

  const startupCandidates = startupEligible.slice(0, maxImmediate);  // 立即执行
  const deferredOverflow = startupEligible.slice(maxImmediate);       // 超出部分延迟

  return {
    candidates: startupCandidates,
    deferredJobs: [
      ...deferredOverflow.map(job => ({ jobId: job.id })),
      ...deferredAgentJobs.map(job => ({
        jobId: job.id,
        delayMs: startupDeferredMissedAgentJobDelayMs ?? 120_000  // 默认 2 分钟
      })),
    ],
  };
}
```

```typescript
export function resolveDefaultCronStaggerMs(expr: string): number | undefined {  // stagger.ts
  // 整点重复表达式自动设置 5 分钟（300000ms）stagger 窗口
  return isRecurringTopOfHourCronExpr(expr) ? DEFAULT_TOP_OF_HOUR_STAGGER_MS : undefined;
}
// DEFAULT_TOP_OF_HOUR_STAGGER_MS = 5 * 60 * 1000 = 300_000
```

**效果示例：**

假设有 3 个 job 都配置了 `"0 * * * *"`（每小时整点），staggerMs=300000：

| Job | SHA-256 偏移 | 实际触发时间 |
|-----|------------|------------|
| job-A | 120000ms | 整点后 2 分钟 |
| job-B | 45000ms | 整点后 45 秒 |
| job-C | 270000ms | 整点后 4.5 分钟 |

同一 job 的偏移在进程重启后保持不变（确定性），不会因重启而改变触发时间。

### 12.3 关键节点详解

#### ① enqueueSystemEvent — 纯内存队列

**文件：** `src/infra/system-events.ts`

```typescript
// system-events.ts — 进程内 Map，按 sessionKey 分桶
const queues = resolveGlobalMap<string, SessionQueue>(SYSTEM_EVENT_QUEUES_KEY);

export function enqueueSystemEvent(text: string, options: SystemEventOptions) {
  const entry = getOrCreateSessionQueue(options.sessionKey);
  entry.queue.push({
    text: cleaned,
    ts: Date.now(),
    contextKey: normalizeContextKey(options.contextKey),
  });
  // 最多保留 20 条，超出时丢弃最旧的
  if (entry.queue.length > MAX_EVENTS) entry.queue.shift();
}
```

**关键特性：**
- 纯内存，不持久化，进程重启后丢失
- 按 `sessionKey` 隔离，不同 session 的事件互不干扰
- 去重：相同 `(text, contextKey, deliveryContext)` 的事件不会重复入队
- 最大 20 条，防止积压

#### ② requestHeartbeat — 叫醒信号

**文件：** `src/infra/heartbeat-wake.ts:314`

```typescript
export function requestHeartbeat(opts: {
  source: HeartbeatWakeSource;   // "cron"
  intent: HeartbeatWakeIntent;   // "immediate" 或 "event"
  reason?: string;               // "cron:jobId"
  sessionKey?: string;
  agentId?: string;
})
```

`requestHeartbeat` 不直接执行任何 Agent 逻辑，它只是向 heartbeat runner 的调度循环发送一个"现在有事要处理"的信号，heartbeat runner 在下一个 tick 检查队列并决定是否真正执行。

#### ③ runHeartbeatOnce — 前置检查

**文件：** `src/infra/heartbeat-runner.ts:1285`

heartbeat runner 在执行前会做一系列检查，任何一个不满足就 skip：

| 检查 | skip 原因 |
|------|----------|
| `cron.enabled = false` | `"disabled"` |
| 不在 `activeHours` 时间窗口内 | `"quiet-hours"` |
| `CommandLane.Main` 有待处理请求 | `"requests-in-flight"` |
| 有 cron job 正在运行 | `"cron-in-progress"` |
| session lane 繁忙 | `"requests-in-flight"` |
| 无任务到期（prompt 为 null） | `"no-tasks-due"` |

**cron 触发时的特殊处理：**

```typescript
// heartbeat-runner.ts:1646
Provider: hasExecCompletion ? "exec-event" : hasCronEvents ? "cron-event" : "heartbeat"
```

当队列中有 cron 事件时，`Provider` 被设为 `"cron-event"`，这个标记会在后续的 session 处理中用于区分"这是系统事件，不是用户交互"，从而不更新 `lastInteractionAt`、不影响 session 的 daily reset 计时。

#### ⑦ drainFormattedSystemEvents — 事件注入 prompt

**文件：** `src/auto-reply/reply/session-system-events.ts`

```typescript
export async function drainFormattedSystemEvents(params: {
  cfg: OpenClawConfig;
  sessionKey: string;
  isMainSession: boolean;
  isNewSession: boolean;
}): Promise<string | undefined> {
  // 从队列取出事件（消费，不是 peek）
  const queued = consumeSelectedSystemEventEntries(
    params.sessionKey,
    selectGenericSystemEvents(peekSystemEventEntries(params.sessionKey)),
  );
  // 格式化为 "System: [时间戳] 文本" 形式
  for (const event of queued) {
    const timestamp = `[${formatSystemEventTimestamp(event.ts, params.cfg)}]`;
    systemLines.push(`System: ${timestamp} ${event.text}`);
  }
  return systemLines.join("\n");
}
```

格式化后的结果（`eventsBlock`）在 `buildPromptPrelude` 中被前置拼接到 prompt：

```typescript
// prompt-prelude.ts
const prependEvents = (body: string) =>
  combinedEventsBlock ? `${combinedEventsBlock}\n\n${body}` : body;
```

**Agent 最终看到的 prompt 结构：**

```
System: [22:00:05] 扫描今天所有出货单的状态，检查是否有异常。

（heartbeat prompt 或空白）
```

### 12.4 与普通对话的差异对比

| 维度 | 普通用户对话 | Cron main session 任务 |
|------|------------|----------------------|
| 入口 | `chat.send` WS → `dispatchInboundMessage` | `enqueueSystemEvent` + `requestHeartbeat` → `runHeartbeatOnce` |
| `Provider` 字段 | 渠道名（`"telegram"`、`"webchat"` 等） | `"cron-event"` |
| prompt 来源 | 用户输入的 `Body` | heartbeat prompt + system event 前置块 |
| `lastInteractionAt` | 更新（算用户交互） | **不更新**（不算用户交互） |
| session daily reset | 可能触发 | **不触发** |
| Pi Runner 执行 | 完全相同 | 完全相同 |
| 工具执行 | 完全相同 | 完全相同 |

**结论：** cron 任务和普通对话在 `runReplyAgent` 之后的所有逻辑（上下文组装、模型调用、工具执行、回复投递）**完全相同**，走同一个 Pi Runner。差异只在"如何把任务内容送进 prompt"这一段。
